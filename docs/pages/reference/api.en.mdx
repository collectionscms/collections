import { RedocStandalone } from 'redoc';
import { useData } from 'nextra/data';

export const getStaticProps = ({ params }) => {
  return fetch(`${process.env.NEXT_PUBLIC_API_ENDPOINT}/api/openapi`)
    .then((res) => res.json())
    .then((repo) => ({
      props: {
        ssg: repo,
      },
    }));
};

export const Redoc = () => {
  const data = useData();
  return <RedocStandalone spec={data} />;
};
